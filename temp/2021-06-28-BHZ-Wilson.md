这里整理一下计算BHZ模型Wilson loop的代码.
{:.info}
<!--more-->
BHZ模型是最初学习拓扑时接触比较早的模型,前面也整理过如何计算BHZ模型的$\mathcal{Z}_2$拓扑不变量,但是其拓扑性质仍然可以通过Wilson loop来进行计算,所以这里就利用Julia来计算一下这个模型的Wilson loop.
# 代码
```julia
using LinearAlgebra,PyPlot,DelimitedFiles
# =====================================================
function Pauli()
    hn::Int64 = 4
    g1 = zeros(ComplexF64,hn,hn)
    g2 = zeros(ComplexF64,hn,hn)
    g3 = zeros(ComplexF64,hn,hn)
    #----------------
    g1[1,1] = 1
    g1[2,2] = -1
    g1[3,3] = 1
    g1[4,4] = -1
    #-------------
    g2[1,2] = 1
    g2[2,1] = 1
    g2[3,4] = -1
    g2[4,3] = -1
    #-------------
    g3[1,2] = -im
    g3[2,1] = im
    g3[3,4] = -im
    g3[4,3] = im
    return g1,g2,g3
end
# =====================================================
function BHZ(kx::Float64,ky::Float64)::Matrix{ComplexF64}
    # 构造系统哈密顿量
    hn::Int64 = 4
    m0::Float64 = 1.5
    tx::Float64 = 1.0
    ty::Float64 = 1.0
    lamx::Float64 = 1.0
    lamy::Float64 = 1.0
    g1 = zeros(ComplexF64,hn,hn)
    g2 = zeros(ComplexF64,hn,hn)
    g3 = zeros(ComplexF64,hn,hn)
    ham = zeros(ComplexF64,hn,hn)
    g1,g2,g3 = Pauli()
    for i1 = 1:hn
        for i2 = 1:hn
            ham[i1,i2] = (m0 - tx*cos(kx) - ty*cos(ky))*g1[i1,i2] + lamx*sin(kx)*g2[i1,i2] + lamy*sin(ky)*g3[i1,i2]
        end
    end
    return ham
end
# ======================================================================
function main()
    nx::Int64 = 100
    ny::Int64 = 800
    Noccu::Int64 = 2  # 占据态数目
    Kx = range(0,2,length = nx)
    Ky = range(-1,0.9999,length = ny)
    klist = []
    Wave = zeros(ComplexF64,4,4,nx)
    Wan = zeros(ComplexF64,Noccu,Noccu)
    ang = zeros(Float64,ny,Noccu)
    f1 = open("test.dat","w")
    for iy in 1:ny
        ky = Ky[iy]*pi
        append!(klist,ky)
        for ix in 1:nx    # 在固定ky的时候，对每一个kx进行对角化
            kx = Kx[ix]*pi
            ham = BHZ(kx,ky)
            val,vec = eigen(ham)
            Wave[:,:,ix] = vec[:,:]  # 存储所有点上的本征矢量
        end
        Wave[:,:,nx] = Wave[:,:,1]  # 首尾相连
        F = 1
        for i1 in 1:nx-1
            for i2 in 1:Noccu
                for i3 in 1:Noccu
                    Wan[i2,i3] = Wave[:,i2,i1]'*Wave[:,i3,i1 + 1]   # 计算Berry联络
                end
            end
            F = F*Wan
        end
        val,vec = eigen(F)
        ang[iy,:] = map(angle,val)/(2*pi)
        writedlm(f1,[ky/pi ang[iy,1] ang[iy,2]])
    end
    close(f1)
    return klist,ang
end
# =================================================================

# plot(k,ang)
@time main()
```

# Slab 结构下的Wilson-Loop
```julia
using LinearAlgebra,PyPlot,DelimitedFiles
# =====================================================
function Pauli()
    hn::Int64 = 4
    g1 = zeros(ComplexF64,hn,hn)
    g2 = zeros(ComplexF64,hn,hn)
    g3 = zeros(ComplexF64,hn,hn)
    #----------------
    g1[1,1] = 1
    g1[2,2] = -1
    g1[3,3] = 1
    g1[4,4] = -1
    #-------------
    g2[1,2] = 1
    g2[2,1] = 1
    g2[3,4] = -1
    g2[4,3] = -1
    #-------------
    g3[1,2] = -im
    g3[2,1] = im
    g3[3,4] = -im
    g3[4,3] = im
    return g1,g2,g3
end
# =====================================================
function BHZ_Slab(on::Int64,kx::Float64,ky::Float64)::Matrix{ComplexF64}
    # 构造系统哈密顿量
    hn::Int64 = 4
    # on::Int64 = 50
    N::Int64 = hn*on
    m0::Float64 = 1.5
    tx::Float64 = 1.0
    ty::Float64 = 1.0
    lamx::Float64 = 1.0
    lamy::Float64 = 1.0
    g1 = zeros(ComplexF64,N,N)
    g2 = zeros(ComplexF64,N,N)
    g3 = zeros(ComplexF64,N,N)
    ham = zeros(ComplexF64,N,N)
    g1,g2,g3 = Pauli()
    for k in 0:on - 1
        if k == 0    # First line     
           for m1 in 1:hn
                for m2 in 1:hn
                    ham[m1,m2] = (m0 - ty*cos(ky))*g1[m1,m2] + lamy*sin(ky)*g3[m1,m2]
                    ham[m1,m2 + hn] = -tx/2.0*g1[m1,m2] + lamx/(2*im)*g2[m1,m2] 
                end
            end
        elseif k == on-1
            for m1 in 1:hn
                for m2 in 1:hn
                    ham[k*hn + m1,k*hn + m2] = (m0 - ty*cos(ky))*g1[m1,m2] + lamy*sin(ky)*g3[m1,m2]
                    ham[k*hn + m1,k*hn + m2 - hn] = -tx/2.0*g1[m1,m2] - lamx/(2*im)*g2[m1,m2]
                end
            end
        else
            for m1 in 1:hn
                for m2 in 1:hn
                    ham[k*hn + m1,k*hn + m2] = (m0 - ty*cos(ky))*g1[m1,m2] + lamy*sin(ky)*g3[m1,m2]
                    #   right hopping
                    ham[k*hn + m1,k*hn + m2 + hn] = -tx/2.0*g1[m1,m2] + lamx/(2*im)*g2[m1,m2]
                    #   left hopping
                    ham[k*hn + m1,k*hn + m2 - hn] = -tx/2.0*g1[m1,m2] - lamx/(2*im)*g2[m1,m2]
                end
            end
        end
    end
    return ham
end
# ======================================================================
function main()
    nx::Int64 = 100
    ny::Int64 = 800
    on::Int64 = 50
    hn::Int64 = 4
    N::Int64 = hn*on
    Nocc::Int64 = Integer(N/2)  # 占据态数目
    Kx = range(0,2,length = nx)
    Ky = range(-1,0.9999,length = ny)
    klist = []
    Wave = zeros(ComplexF64,N,N,nx)
    Wan = zeros(ComplexF64,Nocc,Nocc)
    ang = zeros(Float64,ny,Nocc)
    f1 = open("test.dat","w")
    for iy in 1:ny
        ky = Ky[iy]*pi
        append!(klist,ky)
        for ix in 1:nx    # 在固定ky的时候，对每一个kx进行对角化
            kx = Kx[ix]*pi
            ham = BHZ_Slab(on,kx,ky)
            val,vec = eigen(ham)
            Wave[:,:,ix] = vec[:,:]  # 存储所有点上的本征矢量
        end
        Wave[:,:,nx] = Wave[:,:,1]  # 首尾相连
        F = 1
        for i1 in 1:nx-1
            for i2 in 1:Nocc
                for i3 in 1:Nocc
                    Wan[i2,i3] = Wave[:,i2,i1]'*Wave[:,i3,i1 + 1]   # 计算Berry联络
                end
            end
            F = F*Wan
        end
        val,vec = eigen(F)
        ang[iy,:] = map(angle,val)/(2*pi)
        # writedlm(f1,[ky/pi ang[iy,1] ang[iy,2]])
        writedlm(f1,[ang[iy,:]])
    end
    close(f1)
    return klist,ang
end
# =================================================================

@time main()
```
