# 2.getrf,getri 矩阵求逆

> getrf 对一个矩阵进行LU分解

> getri 计算由LU分解后矩阵的逆 

```fortran
program main
use lapack95	!调用函数库，确保使用的函数存在
!如果时用的是intel fortran，则在编译的时候注释上一行，加入-mkl编译选项即可。
implicit none
integer, parameter :: n = 3
integer :: i, j, temp(n)
real(kind=8) :: a(n,n), aa(n,n)

call random_seed()
call random_number(a)
aa = a

write(*,'(1x,a)') "a = "
do i = 1, n
        write(*,'(*(f12.6,3x))') a(i,:)
end do

!使用库函数求逆
call getrf( a, temp )!先进性LU分解
call getri( a, temp )!然后进行矩阵求逆

write(*,'(1x,a)') 'inv(a) = '
do i = 1, n
        write(*,'(*(f12.6,3x))') a(i,:)
end do

write(*,'(1x,a)') "checking..."
aa = matmul(aa,a)  !// 原矩阵与其逆矩阵的结果为单位矩阵
do i = 1, n
        write(*,'(*(f12.6,3x))') aa(i,:)
end do

end program main
```
